### Oracle Setup Requirements. 
- create new user and grant privileges;
```sql
create user nba identified by ddadmr12;
grant
    connect, 
    alter any index,
      alter any indextype,
      alter any procedure,  
      alter any sequence,  
      alter any table,  
      alter any trigger,  
      alter database,  
      alter rollback segment,  
      alter session,  
      alter system,  
      alter tablespace,  
      alter user,
    create any sequence,  
      comment any table,  
    create any index,  
   create any indextype,  
   create any procedure,  
   create any table,  
   create any trigger,
   create any type,  
   create any view,
    create any synonym,
    create public synonym,      
   create rollback segment,
    create sequence,  
    create session,  
    create table,  
    create tablespace,  
    create trigger,  
    create type,  
    create user,  
    create view,          
    delete any table,
   drop any index,  
   drop any indextype,  
   drop any sequence,
    drop any synonym,  
   drop any table,  
   drop any trigger,  
   drop any type,  
   drop any view,  
   drop rollback segment,  
   drop tablespace,  
   drop user,
    drop public synonym,  
   manage tablespace,  
   restricted session,  
   select any dictionary,  
   select any sequence,  
   select any table,  
   unlimited tablespace
to nba identified by ddadmr12;
```
